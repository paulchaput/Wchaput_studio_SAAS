       ```typescript
       export async function getProjectWithLineItems(id: string) {
         const supabase = await createClient()
         const { data, error } = await supabase
           .from('projects')
           .select(`
             *,
             line_items (
               id, descripcion, referencia, dimensiones,
               cantidad, costo_proveedor, margen, proveedor_id, created_at,
               suppliers ( id, nombre )
             )
           `)
           .eq('id', id)
           .single()
         if (error) throw error
         return data
       }
       ```
       Filter is .eq('id', id) — always filter by project — NEVER return all line_items globally.

    2. Create lib/actions/line-items.ts with three 'use server' functions:

       Zod schema for line items:
       ```typescript
       const lineItemSchema = z.object({
         project_id: z.string().uuid(),
         descripcion: z.string().min(1, 'La descripción es requerida'),
         referencia: z.string().optional(),
         dimensiones: z.string().optional(),
         cantidad: z.coerce.number().int().positive('La cantidad debe ser mayor a 0'),
         proveedor_id: z.string().uuid().optional().nullable(),
         costo_proveedor: z.coerce.number().nonnegative('El costo no puede ser negativo'),
         // User enters "50" for 50% — CRITICAL: transform to 0.50 for DB storage
         margen: z.coerce.number().min(0).max(99).transform(v => v / 100),
       })
       ```

       createLineItemAction(formData: FormData): Promise<{ error?: string }>
       - Parse with lineItemSchema.safeParse(Object.fromEntries(formData))
       - If invalid: return { error: parsed.error.errors[0].message }
       - supabase.from('line_items').insert(parsed.data)
       - On success: revalidatePath('/proyectos/'+parsed.data.project_id), return {}
       - On DB error: return { error: error.message }
       - NOTE: Do NOT redirect — caller stays on project detail page

       updateLineItemAction(lineItemId: string, formData: FormData): Promise<{ error?: string }>
       - Same Zod schema (project_id passed via hidden input)
       - supabase.from('line_items').update(parsed.data).eq('id', lineItemId)
       - On success: revalidatePath('/proyectos/'+parsed.data.project_id), return {}

       deleteLineItemAction(lineItemId: string, projectId: string): Promise<{ error?: string }>
       - supabase.from('line_items').delete().eq('id', lineItemId)
       - On success: revalidatePath('/proyectos/'+projectId), return {}
       - Simple — no Zod needed for delete

    Critical: The margen Zod transform (v => v / 100) is the ONLY place where percent→decimal conversion happens. Never do this conversion in components. Never store 50 when 0.50 is expected.
  </action>
